CREATE TABLE userlist(
	id INT  AUTO_INCREMENT,
	 name VARCHAR(20),
  	 usename VARCHAR(20) UNIQUE,
  	 address VARCHAR(100),
  	 tel VARCHAR(20),
  	 gender VARCHAR(5),
  	 birthday DATE,
  	 pass VARCHAR(256) NOT NULL,
  	PRIMARY KEY (id)
);




CREATE TABLE contributions(
	id INT  AUTO_INCREMENT,
	 filename VARCHAR(50) ,
  	 usename VARCHAR(20),
  	uploaddate timestamp not null default current_timestamp,
  	PRIMARY KEY (id)
);
